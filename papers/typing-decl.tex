\def\WfEmpty{
  \UAX{WfEmpty}
  {}
  {$\seq []$}
  {}
}  
  
\def\WfVar{
  \UAX{WfVar}
  {$`G \type A : s$}
  {$\seq `G, x : A$}
  {$s `: \setproptype{} `^{} x `; `G$}
}

\def\PropSet{
  \UAX{PropSet}
  {$\wf `G$}
  {$`G \seq s : \Type$}
  {$s `: \{ \Prop, \Set \}$} 
}

\def\TypeType{
  \UAX{Type}
  {$\wf `G$}
  {$`G \seq \Type(i) : \Type(i + 1)$}
  {$i `: `N$}
}

\def\Var
{\BAX{Var}
  {$\wf `G$}
  {$x : A `: `G$}
  {$`G \seq x : A$}
  {}
}

\def\Prod{
  \BAX{Prod}
  {$`G \seq T : s_1$}
  {$`G, x : T \seq U : s_2$}
  {$`G \seq \Pi x : T.U : s_3$}
  {$(s_1, s_2, s_3) `: \mathcal{R} `^{} x `; `G$}
}

\def\Abs{
  \BAX{Abs}
  {$`G \seq \Pi x : T. U : s $}
  {$`G, x : T \seq M : U $}
  {$`G \seq \lambda x : T. M : \Pi x : T.U$}
  {$x `; `G$}
}

\def\App{
  \BAX{App}
  {$`G \seq f : \Pi x : V. W $}
  {$`G \seq u : V $}
  {$`G \seq (f u) : W [ u / x ]$}
  {$$}
}

\def\LetIn{
  \BAX{LetIn}
  {$`G \seq t : T $}
  {$`G, x : T \seq v : V $}
  {$`G \seq \letml~x = t~\inml~v : V[t / x]$}
  {$x `; `G$}
}

\def\SigmaR{
  \BAX{Sigma}
  {$`G \seq T : s $}
  {$`G, x : T \seq U : s $}
  {$`G \seq \Sigma x : T.U : s $}
  {$s `: \{ \Prop, \Set \} `^{} x `; `G$} 
}

\def\Sum{
  \TAX{Sum}
  {$`G \seq \Sigma x : T.U : s$}
  {$`G \seq t : T $}
  {$`G \seq u : U[t/x] $}
  {$`G \seq (x := t, u : U) : \Sigma x : T.U$}
  {}
}
 
\def\LetSum{
  \BAX{LetSum}
  {$`G \seq t : `S x : T. U $}
  {$`G, x : T, u : U \seq v : V $}
  {$`G \seq \letml~(x, u) = t~\inml~v : V$}
  {$x,y `; `G$}
}

\def\Subset{
  \BAX{Subset}
  {$`G \seq U : \Set$}
  {$`G, x : U \seq P : \Prop $}
  {$`G \seq \subset{x}{U}{P} : \Set$}
  {$x `; `G$}
}

\def\LetSub{
  \UAX{LetSub}
  {$`G \seq t : \subset{x}{T}{P}$}
  {$`G \seq t : T$}
  {}
}

\def\Subsum{
  \TAX{Subsum}
  {$`G \seq T : s$}
  {$`G \seq t : U$}
  {$U \sub T$} % `G \subt 
  {$`G \seq t : T$}
  {}
}

\def\Coerce{\Subsum}

\def\Conv{
  \TAX{Conv}
  {$`G \seq T : s$}
  {$`G \seq t : U$}      
  {$U \eqbi T$}
  {$`G \seq t : T$}
  {}
}
