\begin{lemma}[Coercion de sortes]
  \label{coerce-sorts}
  Si $\subimpl{`G}{e}{s}{T}$ ou $\subimpl{`G}{e}{T}{s}$ alors $T
  \eqbr s$ et $e = \ctxdot$.
\end{lemma}
\begin{proof}  
  Clairement on ne peut dériver $s \suba T$ que par \irule{SubConv}
  (éventuellement précédé de \irule{SubHnf}). En effet seule la règle
  \irule{SubProof} pourrait s'appliquer, mais cela impliquerait que
  $T \eqbr \mysubset{x}{U}{P}$ avec $s \suba U$ et ainsi de suite. La
  seule possibilité est de dériver $s \eqbr T$ ou $s \eqbr U$, auquel cas $U$ est une
  sorte ce qui contredit le fait que $\mysubset{x}{U}{P} : s$ dans le
  cas précédent. On dérive donc $s \suba T$ si et seulement si $s
  \eqbr T$.
\end{proof}

\begin{definition}[Coercion de contextes]
  \label{coercion-ctx}
  On définit inductivement la coercion de deux contextes de coercions
  algorithmiques par les règles suivantes:
  \begin{itemize}
  \item $[] \suba []$
  \item $(`G, x : T) \suba (`G', x : T')$ si $`G \suba `G'$ et $T \suba T'$.
  \end{itemize}
\end{definition}

De même pour les coercions explicites dérivées par le jugement $\subimpl{`G}{c}{T}{S}$.
\begin{definition}[Coercion explicites de contextes]
  \label{coercion-ctx-i}
  On définit inductivement la coercion de deux contextes de coercions
  explicites par les règles suivantes:
  \begin{itemize}
  \item $[] \suba []$
  \item $`r, c : (`G, x : T) \suba (`G', x : T')$ si $`r : `G \suba `G'$ et 
    $\subimpl{`G}{c}{T}{T'}$.
  \end{itemize}
\end{definition}

Clairement toute coercion de contexte algorithmique correspond à une
coercion de contexte explicite et vice-versa.

\begin{definition}[Extension de la substitution aux coercions de
  contextes]
  \label{subst-coercion-ctx}
  On définit la substitution d'une coercion de contexte inductivement:
  \begin{itemize}
  \item $t[[]] = t$
  \item $t[(`r, c) : (`G, x : T) \suba (`G', x : T')] = t[`r : `G \suba `G'][c[x]/x]$
  \end{itemize}
\end{definition}

\begin{lemma}[Stabilité par affaiblissement des coercions]
  \label{coerce-narrowing-coercion}
  Si $\subimpl{`G}{c}{T}{T'}$ et $\ip{T'}{`D} `=
  \ip{T'}{`G}[`r]$, alors pour tout $`D$ tel que $`r : `D \sub `G$,
  $\subimpl{`D}{c'}{T}{T'}$ et $c' `= c[`r]$ avec une dérivation de même
  taille.
\end{lemma}

\begin{proof}
  Par induction sur la dérivation de coercion:

  \begin{induction}

    $\subimpl{`D}{c[`r]}{\hnf{T}}{\hnf{T'}}$. 
    On peut donc dériver $\subimpl{`D}{c[`r]}{T}{T'}$ par \irule{SubHnf}.

    \case{SubConv}
    On a $T \eqbr T'$ et $c = \ctxdot$.

    on a bien $c' `= c[`r]$.

        {`=}{\lambda x : \ip{C}{`D}.(c_2[`r])[\ctxdot~(c_1[`r])[x]]}
        
        \step{Coercion identité dans $`s$}
        {=}{\lambda x : \ip{C}{`D}.(c_2[`s])[\ctxdot~(c_1[`r])[x]]}
      \end{array}
      \]
      
    \case{SubSigma}\quad
    \begin{prooftree}
      \BAX{}
      {$\subimpl{`G}{c_1}{A}{C}$}
      {$\subimpl{`G, x : A}{c_2}{B}{D}$}
      {$\subimpl{`G}{\pair{\ip{\Sigma x : C.D}{`G}}{c_1[\pi_1~\ctxdot]}{c_2[\pi_1~\ctxdot/x][\pi_2~\ctxdot]}}
        {\Sigma x : A.B}{\Sigma x : C.D}$}
      {}
    \end{prooftree}
    Par induction on a $\subimpl{`D}{c_1[`r]}{A}{C}$. On peut définir
    la coercion $`s = `r, \ctxdot : (`D, x : A) \suba (`G, x : A)$ et obtenir par
    induction:
    $\subimpl{`D, x : A}{c_2[`s]}{B}{D}$
    
    On peut alors appliquer \irule{SubSigma} pour obtenir:
    \[\subimpl{`D}
    {c' = \pair{\ip{\Sigma x : C.D}{`D}}{(c_1[`r])[\pi_1~\ctxdot]}{(c_2[`s])[\pi_1~\ctxdot/x][\pi_2~\ctxdot]}}
    {\Sigma x : A.B}{\Sigma x : C.D}\]
    
    On a:
    \[\begin{array}{ll}
      \firsteq{c[`r]}

      \step{Définition}
      {\eqdef}
      {(\pair{\ip{\Sigma x : C.D}{`G}}{c_1[\pi_1~\ctxdot]}{c_2[\pi_1~\ctxdot/x][\pi_2~\ctxdot]})[`r]}
      
      \step{Définition de la substitution, $x `; `r$}
      {=}{\pair{\ip{\Sigma x : C.D}{`G}[`r]}{c_1[`r][\pi_1~\ctxdot]}{c_2[`r][\pi_1~\ctxdot/x][\pi_2~\ctxdot]}}

      \step{Condition}
      {=}{\pair{\ip{\Sigma x : C.D}{`D}}{c_1[`r][\pi_1~\ctxdot]}{c_2[`r][\pi_1~\ctxdot/x][\pi_2~\ctxdot]}}

      \step{Coercion identité dans $`s$}
      {=}{\pair{\ip{\Sigma x :
            C.D}{`D}}{c_1[`r][\pi_1~\ctxdot]}{c_2[`s][\pi_1~\ctxdot/x][\pi_2~\ctxdot]}}

      \step{Définition}{\eqdef}{c'}
      \end{array} 
      \]
      
   \case{SubProof}\quad
   \begin{prooftree}
     \UAX{}
     {$\subimpl{`G}{d}{T}{U}$}
     {$\subimpl{`G}
       {c = \elt{\ip{U}{`G}}
         {\ip{(\lambda x : U.P)}{`G}}
         {d}
         {\ex{\ipG{`G}}{\ip{P}{`G, x : U}[d/x]}}}
       {T}{\mysubset{x}{U}{P}}$}
     {}
   \end{prooftree}

   Par induction, on a $\subimpl{`D}{d[`r]}{T}{U}$. On peut donc dériver:
   \[\subimpl{`D}{c' = \elt{\ip{U}{`D}}
     {\ip{(\lambda x : U.P)}{`D}}
     {d[`r]}
     {\ex{\ipG{`D}}{\ip{P}{`D, x : U}[d[`r]/x]}}}
   {T}{\mysubset{x}{U}{P}}\]
   
   On a bien $c[`r] `= c'$, car:
   \begin{equations}
     \firsteq{c[`r]}

     \step{Définition}
     {\eqdef}{(\elt{\ip{U}{`G}}
       {\ip{(\lambda x : U.P)}{`G}}
       {d}
       {\ex{\ipG{`G}}{\ip{P}{`G, x : U}[d/x]}})[`r]}

     \step{Condition $\ip{\mysubset{x}{U}{P}}{`D} `= \ip{\mysubset{x}{U}{P}}{`G}[`r]$}
     {`=}{\elt{\ip{U}{`D}}
       {\ip{(\lambda x : U.P)}{`D}}
       {d[`r]}
       {\ex{\ipG{`G}}{\ip{P}{`G, x : U}[d/x][`r]}}}
     
     \step{Indifférence aux preuves}
     {=_\sigma}{\elt{\ip{U}{`D}}
       {\ip{(\lambda x : U.P)}{`D}}
       {d[`r]}
       {\ex{\ipG{`D}}{\ip{P}{`D, x : U}[d[`r]/x]}}}
     
     \step{Définition}
     {\eqdef}{c'}
   \end{equations}

   \case{SubSub} Direct par induction.
  \end{induction}
\end{proof}

\begin{lemma}[Stabilité par affaiblissement]
  \label{coerce-narrowing-term}
  Si $`r : `D \sub `G$, $\talgo{`G}{t}{T}$ alors 
  $\talgo{`D}{t}{T'}$ et il existe $`a$, $\subimpl{`D}{`a}{T'}{T}$ avec
  $\ip{t}{`G}[`r] `= `a[\ip{t}{`D}]$.
\end{lemma}

\begin{proof}
  La première partie du lemme se déduit par applications répétées du
  lemme de restriction \ref{narrowing-a}.
  
  On va utiliser à plusieurs reprises le résultat suivant:
  Si une prémisse du jugement sur lequel on fait la récurrence est de la
  forme $\timpl{`G}{T}{s}$, alors on peut appliquer l'hypothèse
  d'induction pour obtenir $\timpl{`D}{T}{s'}$ avec $`a$ tel que
  $\subimpl{`D}{`a}{s}{s'}$ et $\ip{T}{`G}[`r] `=
  `a[\ip{T}{`D}]$. Or comme $s$ est une sorte, on a $s' = s$ et $`a =
  \ctxdot$. On a donc $\ip{T}{`G}[`r] `= \ip{T}{`D}$. 

  Par induction sur la dérivation de $t$.
  
  \begin{induction}
    \case{Var} 
    On a:
    \typenva
    \begin{prooftree}
      \BAX{Var}
      {$\wf `G$}
      {$y : T `: `G$}
      {$`G \seq y : T$}
      {}
    \end{prooftree}    
    
    Par définition de la coercion de contextes, il existe $c : (T' \sub
    T) `: `r$. Soit $`a = c$, on a bien: $\ip{y}{`G}[`r] = c[y] =
    `a[\ip{y}{`D}]$.
    
    \case{App}
    \typenva
    On a: 
    \begin{prooftree}
      \TAX{App}
      {$\talgo{`G}{f}{F} \quad \mualgo(F) = \Pi y : A. B$}
      {$\talgo{`G}{e}{E} \quad \talgo{`G}{E, A}{s}$}
      {$E \sub A $}
      {$\talgo{`G}{(f~e)}{B[ e / y ]}$}
      {}
    \end{prooftree}
    
    et donc par induction, $\talgo{`D}{f}{F'}$ avec
    $\subimpl{`D}{`a_f}{F'}{F}$ et
    $\talgo{`D}{e}{E'}$ avec $\subimpl{`D}{`a_e}{E'}{E}$.

    Par définition de l'interprétation, 
    \begin{eqnarray*}
      \ip{(f~e)}{`G}
      & = & (((\pi_F[\ip{f}{`G}])~(c_e[\ip{e}{`G}])) \\
      \text{ où } & & \\
      \pi_F & = & \coerce{`G}{F}{\Pi y : A.B} \\
      c_e & = & \coerce{`G}{E}{A}.
    \end{eqnarray*}
    
    De même:
    \begin{eqnarray*}
      \ip{f~e}{`D}
      & = & (\pi_{F'}[\ip{f'}{`D}])~(c_{e'}[\ip{e'}{`D}]) \\
      \text{ où } & & \\
      \pi_{F'}  & = & \coerce{`D}{F'}{\Pi y : A'.B'} \\
      c_{e'} & = & \coerce{`D}{E'}{A'}
    \end{eqnarray*}

    On a $\subimpl{`G}{\pi_F}{F}{\Pi y : A.B}$, donc
    par le lemme \ref{coerce-narrowing-coercion} on obtient
    $\subimpl{`D}{\pi_F[`r]}{F}{\Pi y : A.B}$. On peut appliquer ce
    lemme puisque $\ip{Pi y : A.B}{`D} `= \ip{\Pi y : A.B}{`G}[`r]$ par
    induction ($`a$ est nécessairement l'identité).

    On a donc les coercions suivantes dans l'environnement $`D$:
    \[
    \xymatrix{
      F\ar[rr]_{\pi_F[`r]} & & 
      \Pi y : A.B \\
      F'\ar[u]^{`a_f}\ar[rr]^{\pi_{F'}} & &
      {\Pi y : A'.B'}\ar@{-->}[u]^{c_f}}
    \]

    Par symétrie et transitivité de la coercion, on en déduit qu'il
    existe $c_f$, $\subimpl{`D}{c_f}{\Pi y : A'.B'}{\Pi y : A.B}$.
    La coercion $c_f$ est nécessairement de la forme:
    $\lambda y : \ip{A}{`D}.c_2[\ctxdot~c_1[y]]$ où 
    $\subimpl{`D}{c_1}{A}{A'}$ et $\subimpl{`D, x : A}{c_2}{B'}{B}$.

    Par le lemme \ref{coerce-narrowing-coercion} on a
    $\subimpl{`D}{c_e[`r]}{E}{A}$, de même que précédemment on obtient
    la condition nécessaire par induction sur la dérivation de $\talgo{`G}{A}{s}$.

    On a donc les coercions suivantes dans l'environnement $`D$:   
    \[
    \xymatrix{
      E\ar[rr]_{c_e[`r]} & & 
      A\ar[d]^{c_1} \\
      E'\ar[u]^{`a_e}
      \ar@{-->}[rr]^{c_{e'}} & & A'}
    \]
       
    
    Par transitivité, il existe donc une coercion $\subimpl{`D}{c_{E',A} `=
      c_e[`r] `o `a_e}{E'}{A}$.    
    Par le lemme \ref{coerce-narrowing-coercion} en utilisant la
    coercion de contexte $\ctxdot, \ldots, c_{E',A} : (`D, x : E') \suba
    `D, x : A$ et la dérivation de $c_2$ on a donc:
    $\subimpl{`D, x : E'}{c_2'}{B'}{B}$ avec $c_2' `= c_2[(c_e[`r])[`a_e[x]]/x]$.
    
    Par le lemme \ref{coerce-subst} avec $`D \typec e : E'$ et la
    dérivation de $c_2'$ on obtient:
    $\subimpl{`D}{c_2''}{B'[e/x]}{B[e/x]}$ avec \[c_2'' `=
    c_2'[\ip{e}{`D}/x] `= c_2[(c_e[`r])[`a_e[\ip{e}{`D}]]/x]\]
  
    Soit $`a = c_2''$, on peut vérifier:
    \[\begin{array}{ll}
      \firsteq{`a[\ip{f~e}{`D}]}
      
      \step{Définition de l'interprétation}
      {\eqdef}{`a[(\pi_{F'}[\ip{f}{`D}])[c_{e'}[\ip{e}{`D}]]]}
      
      \step{Définitions de $`a$ et $c_{e'}$}
      {`=}{c_2[c_e[`r]~[(`a_e[\ip{e}{`D}])]/x]~[
        ((\pi_{F'}[\ip{f}{`D}])~[(c_1 `o c_e[`r] `o `a_e)[\ip{e}{`D}]])]}
      
      \step{Définition de la composition}
      {=}{c_2[c_e[`r]~[(`a_e[\ip{e}{`D}])]/x]~[
        ((\pi_{F'}[\ip{f}{`D}])~[c_1[c_e[`r][`a_e[\ip{e}{`D}]]]])]}

      \step{Définition de $c_f$ ($= \lambda x.c_2[\ctxdot~c_1[x]]$)}
      {=}{c_f[(\pi_{F'}[\ip{f}{`D}])~[c_e[`r]~[`a_e[\ip{e}{`D}]]]]}
      
      \step{Commutation du diagramme pour la fonction}
      {`=}{(\pi_F[`r][`a_f[\ip{f}{`D}]]~[c_e[`r][`a_e~\ip{e}{`D}]])}
      
      \step{Définitions de $`a_f$ et $`a_e$}
      {`=}{(\pi_F[`r][\ip{f}{`G}[`r]])~[c_e[`r][\ip{e}{`G}[`r]]]}

      \step{Définition de l'interprétation}{\eqdef}{\ip{f~e}{`G}[`r]}
      \vspace{0.2em}
    \end{array}\]
    
    \vspace{1cm}
    
    \TODO{Autres cas}
  \end{induction}
\end{proof}
