\begin{lemma}[Coercion de sortes]
  \label{coerce-sorts}
  Si $\subimpl{`G}{e}{s}{T}$ ou $\subimpl{`G}{e}{T}{s}$ alors $T
  \eqbr s$ et $e = \ctxdot$.
\end{lemma}
\begin{proof}  
  Clairement on ne peut dériver $s \suba T$ que par \irule{SubConv}
  (éventuellement précédé de \irule{SubHnf}). En effet seule la règle
  \irule{SubProof} pourrait s'appliquer, mais cela impliquerait que
  $T \eqbr \mysubset{x}{U}{P}$ avec $s \suba U$ et ainsi de suite. La
  seule possibilité est de dériver $s \eqbr T$ ou $s \eqbr U$, auquel cas $U$ est une
  sorte ce qui contredit le fait que $\mysubset{x}{U}{P} : s$ dans le
  cas précédent. On dérive donc $s \suba T$ si et seulement si $s
  \eqbr T$.
\end{proof}

\begin{lemma}[Affaiblissement et interprétation]
  \label{interp-weak}
  Si $\talgo{`G}{t}{T}$ alors pour tout $`D ``=) `G$, $\ip{t}{`G} =
  \ip{t}{`D}$.
\end{lemma}
\begin{proof}
  Les seuls endroits où les environnement sont utilisés dans
  l'interprétation est lors des appels à la fonction de typage
  algorithmique, or on a bien la propriété que si $\talgo{`G}{t}{T}$
  alors $\talgo{`D}{t}{T}$ quand $`G ``(= `D$ par affaiblissement. Cette
  propriété est donc bien vérifiée.
\end{proof}

\begin{lemma}[Stabilité de la coercion par substitution]
  \label{coerce-subst}
  Si $\talgo{\GD}{T, T'}{s}$, $\talgo{`G}{u}{U}$, alors
  $\begin{array}{lcl}
    \wf{\GD} & "=>" & \ipG{\Gr} `= \ipG{\GD}[\ip{u}{`G}/x] \\
    \talgo{\GD}{t}{T} & "=>" & \ip{t[u/x]}{\Gr} `= \ip{t}{\GD}[\ip{u}{`G}/x] \\
    \subimpl{\GD}{c}{T}{T'} 
    & "=>" & \subimpl{\Gr}{c'}{T[u/x]}{T'[u/x]}
    `^ c' `= c[\ip{u}{`G}/x]
  \end{array}$
\end{lemma}

\begin{proof}
  Par induction mutuelle sur les dérivations de bonne formation, typage
  et coercion.
  
  \begin{induction}
    \case{WfEmpty} Trivial.

    \case{WfVar} Par induction sur la longueur de $`D$.
    \begin{induction}
      \caseother{$`D = []$} Alors on a:
      \begin{prooftree}
        \UAX{}
        {$\tgen{`G}{U}{s}$}
        {$\wf `G, x : U$}
        {$s `: \setproptype{} `^{} x `; `G$}
      \end{prooftree}
      Clairement $\ipG{\Gr} = \ipG{`G} = \ipG{`G, x : U}[\ip{u}{`G}/x]$
      puisque $x `; `G$.

      \caseother{$`D = `D', x : A$}
      Alors on a:
      \begin{prooftree}
        \AXC{$\talgo{`G, x : U, `D'}{A}{s}$}
        \UIC{$\wf{`G, x : U, `D', y : A}$}
      \end{prooftree}
      
      Par induction on a \[\ip{A[u/x]}{`G, `D'[u/x]} `=
      \ip{A}{`G, x : U, `D'}[\ip{u}{`G}/x] `^ \ipG{`G, x : U,
        `D'}[\ip{u}{`G}/x] `= \ipG{`G, `D'[u/x]}\] donc
      \begin{eqnarray*}
        \ipG{\Gr} & = & \ipG{`G, `D'[u/x], A'[u/x]} \\
        & \eqdef & \ipG{`G, `D'[u/x]}, \ip{A'[u/x]}{`G, `D'[u/x]} \\
        & = & \ipG{`G, x : U, `D'}[\ip{u}{`G}/x], \ip{A}{`G, x : U,
          `D'}[\ip{u}{`G}/x] \\
        & = & \ipG{`G, x : U, `D', A}[\ip{u}{`G}/x] \\
        & = & \ipG{\GD}[\ip{u}{`G}/x]        
      \end{eqnarray*}
    \end{induction}
    
    \case{SubHnf} On a: 
    \begin{prooftree}
      \AXC{$\subimpl{\GD}{c}{\hnf{T}}{\hnf{T'}}$}
      \UIC{$\subimpl{\GD}{c}{T}{T'}$}
    \end{prooftree}
    
    \def\as{\overrightarrow{a}}    
    \def\bs{\overrightarrow{b}}    
    \def\asux{\overrightarrow{a[u/x]}}
    \def\bsux{\overrightarrow{b[u/x]}}
    \def\ipux{[\ip{u}{`G}/x]}

    Par induction on a
    $\subimpl{\Gr}{c'}{\hnf{T}[u/x]}{\hnf{T'}[u/x]}$ avec $c' `=
    c\ipux$.
    Si $\hnf{T[u/x]} = \hnf{T}[u/x]$ et $\hnf{T'[u/x]} = \hnf{T'}[u/x]$
    c'est direct par induction. Sinon, on a $\hnf{T} = x~\as$ ou $\hnf{T'} =
    x~\as$. Les deux cas sont similaires, on traite le cas ou $\hnf{T} =
    x~\as$. Le jugement $x~\as \suba \hnf{T'}$
    ne peut être dérivé que par \irule{SubProof} ou
    \irule{SubConv}. 

    \begin{induction}
      \case{SubProof} On a:
      \begin{prooftree}
        \AXC{$\subimpl{\GD}{d}{x~\as}{U'}$}
        \UIC{$\subimpl{\GD}{c}{x~\as}{\hnf{T'} = \mysubset{y}{U'}{P}}$}
      \end{prooftree}
      avec \[c = \elt{\ip{U'}{\GD}}{\ip{\lambda x : U'.P}{\GD}}{d}
      {\ex{\ip{P}{\GD, y : U'}[d/y]}}\]

      Par induction on a donc une dérivation de
      $\subimpl{\Gr}{d'}{u~\asux}{U'[u/x]}$ avec $d ' `=
      d\ipux$. 
      Par le lemme \ref{coerce-hnf} on a aussi une dérivation de 
      $\subimpl{\Gr}{d''}{\hnf{(u~\asux)}}{\hnf{(U'[u/x])}}$ avec $d'' `=
      d'\ipux$. 
      
      Par le lemme \ref{coerce-sorting} on a
      $\timpl{\GD}{\mysubset{y}{U'}{P}}{\Set}$, et par inversion
      $\timpl{\GD}{U'}{\Set}$ et $\timpl{\GD, y : U'}{P}{\Prop}$. On
      peut donc appliquer l'hypothèse d'induction pour obtenir:
      $\ip{U'[u/x]}{\Gr} `= \ip{U'}{\GD}\ipux$ et
      $\ip{P[u/x]}{\Gr, y : U'[u/x]} `= \ip{P}{\GD, y : U'}\ipux$. 
      Par substitutivité on a aussi $\timpl{\Gr}{\mysubset{y}{U'[u/x]}{P[u/x]}}{\Set}$.
      On peut donc dériver:    
      \begin{prooftree}
        \AXC{$\subimpl{\Gr}{d'}{\hnf{(u~\asux)}}{\hnf{(U'[u/x])}}$}
        \UIC{$\subimpl{\Gr}{d'}{\hnf{(u~\asux)}}{U'[u/x]}$}
        \UIC{$\subimpl{\Gr}{c'}{\hnf{T[u/x]}}{\hnf{(T'[u/x])} = \mysubset{y}{U'[u/x]}{P[u/x]}}$}
        \UIC{$\subimpl{\Gr}{c'}{T[u/x]}{T'[u/x]}$}
      \end{prooftree}
      avec \[c' = \elt{\ip{U'[u/x]}{\Gr}}{\ip{\lambda y : U'[u/x].P[u/x]}{\Gr}}{d'}
      {\ex{\ip{P[u/x]}{\Gr, y : U'[u/x]}[d'/y]}}\]

      On a aussi:
      \begin{eqnarray*}
        \ip{P[u/x]}{\Gr, y : U'[u/x]}[d'/y] & 
        `= & \ip{P}{\GD, y : U'}\ipux[d'/y] \\
        & `= & \ip{P}{\GD, y : U'}\ipux[d\ipux/y] \\
        & = & \ip{P}{\GD, y : U'}[d/y]\ipux
      \end{eqnarray*}

      Soit \[A \eqdef \ip{P[u/x]}{\Gr, y : U'[u/x]}[d'/y] `^ B \eqdef
      \ip{P}{\GD, y : U'}[d/y]\] on a donc:
      \begin{eqnarray*}
        c' & \eqdef &
        \elt{\ip{U'[u/x]}{\Gr}}
        {\ip{(\lambda y : U'[u/x].P[u/x])}{\Gr}}
        {d'}
        {\ex{A}} \\
        & `= &
        \elt{\ip{U'[u/x]}{\Gr}}
        {\ip{\lambda y : U'[u/x].P[u/x]}{\Gr}}
        {d\ipux}
        {\ex{A}} \\
        & `= &
        \elt{\ip{U'}{\GD}\ipux}
        {\ip{\lambda y : U'.P}{\GD}\ipux}
        {d\ipux}
        {(\ex{B})\ipux} \\
        & = &
        (\elt{\ip{U'}{\GD}}{\ip{\lambda y : U'.P}{\GD}}{d}
        {\ex{B}})\ipux \\
        & = & c\ipux
      \end{eqnarray*}
      
      \case{SubConv}
      Alors on a $\hnf{T} = x~\as$ et $\hnf{T'} = x~\bs$ où $\as `=
      \bs$. Par substitutivité de l'équivalence, $\hnf{T}[u/x] `=
      \hnf{T'}[u/x]$, donc par le lemme
      \ref{coerce-eqbr-eq-id}, on a
      $\subimpl{\Gr}{c'}{u~\asux}{u~\bsux}$ est dérivable et $c' `=
      \ctxdot = \ctxdot\ipux$.
    \end{induction}    
    
    Dans le cas où $\hnf{T'} = x~\as$ et $\hnf{T} "/=" x~\as$, le jugement
    $\subimpl{\GD}{c}{\hnf{T}}{x~\as}$ ne peut être dérivé que par 
    \irule{SubSub} ou \irule{SubConv}.
    \begin{induction}
      \case{SubSub} On a:
      \begin{prooftree}
        \AXC{$\subimpl{\GD}{c'}{U'}{x~\as}$}
        \UIC{$\subimpl{\GD}{c = c'[\eltpit~\ctxdot]}{\hnf{T} = \mysubset{y}{U'}{P}}{x~\as}$}
      \end{prooftree}
      
      Par induction, on a une dérivation de
      $\subimpl{\Gr}{c''}{U'[u/x]}{u~\asux}$ avec $c'' `= c'\ipux$.
      On peut donc dériver:
      \begin{prooftree}
        \AXC{$\subimpl{\Gr}{c''}{U'[u/x]}{u~\asux}$}
        \UIC{$\subimpl{\Gr}{c''[\eltpit~\ctxdot]}
          {\hnf{T[u/x]} =
            \mysubset{y}{U'[u/x]}{P[u/x]}}{\hnf{T'[u/x]}}$}
        \UIC{$\subimpl{\Gr}{c''[\eltpit~\ctxdot]}{T[u/x]}{T'[u/x]}$}
      \end{prooftree}      
      
      Clairement, $c''[\eltpit~\ctxdot] `= c'\ipux[\eltpit~\ctxdot] =
      c'[\eltpit~\ctxdot]\ipux = c\ipux$.
    \end{induction}
    

    \case{SubConv}
    On a $T = \hnf{T}$, $T' = \hnf{T'}$, $T \eqbr T'$ et $c `= \ctxdot$. 
    Par substitutivité de la $\beta\rho$-équivalence, on a aussi $T[u/x] \eqbr
    T'[u/x]$. 
    Par le lemme \ref{coerce-eqbr-eq-id}, on
    sait qu'il existe une coercion $c'$ telle que le jugement
    $\subimpl{\Gr}{c'}{T[u/x]}{T'[u/x]}$ est dérivable et $c' `= \ctxdot$.
    On a bien $c' `= c\ipux$.
    
    \case{SubProd}
    On a:
    \begin{prooftree}
      \AXC{$\subimpl{\GD}{c_1}{A'}{A}$}
      \AXC{$\subimpl{\GD, y : A'}{c_2}{B}{B'}$}
      \BIC{$\subimpl{\GD}
        {\lambda y : \ip{A'}{\GD}.~c_2[\ctxdot~c_1[y]]}
        {\Pi y : A.B}{\Pi y : A'.B'}$}
    \end{prooftree}
    
    Par induction et application de \irule{SubProd}:
    \begin{prooftree}
      \AXC{$\subimpl{\Gr}{c_1'}{A'[u/x]}{A[u/x]}$}
      \AXC{$\subimpl{\Gr, y : A'[u/x]}{c_2'}{B[u/x]}{B'[u/x]}$}
      \BIC{$\subimpl{\Gr}{c'}
        {\Pi y : A[u/x].B[u/x]}{\Pi y : A'[u/x].B'[u/x]}$}
    \end{prooftree}
    
    où \[c' = \lambda y : \ip{A'[u/x]}{\Gr}.~c_2'[\ctxdot~c_1'[y]] \quad c_1'
    `= c_1\ipux \quad c_2' `= c_2\ipux\]

    Par le lemme de coercion et sortage (\ref{coerce-sorting}), avec
    $\talgo{\GD}{\Pi y : A'.B', \Pi y : A.B}{s}$ on a
    $\talgo{\GD}{A', A}{s_1}$. On applique l'hypothèse d'induction
    pour obtenir $\ip{A'[u/x]}{\Gr} `= \ip{A'}{\GD}\ipux$.
    On a donc bien $c' `= c\ipux$. 
    
    \caseother{\irule{SubSigma}, \irule{SubProof}, \irule{SubSub}} Idem, direct par induction.
    
    \case{PropSet} Trivial.
    
    \case{Var} 
    On a:
    \typenva
    \begin{prooftree}
      \BAX{}
      {$\wf \GD$}
      {$y : T `: (\GD)$}
      {$\GD \seq y : T$}
      {}
    \end{prooftree}    
    \typenvi

    \begin{induction}
      \caseother{} Si $x "/=" y$, alors par définition de l'interprétation,
      on doit montrer
      $\ip{y[u/x]}{\Gr} = y = \ip{y}{\GD}\ipux$.
      
      \caseother{} Sinon, $t[u/x] = u$ et $\ip{u}{\Gr} = \ip{x}{\GD}\ipux =
      \ip{u}{\GD}$. On utilise ici le fait que $\ip{t}{`G} = \ip{t}{`D}$
      si $\talgo{`G}{t}{T}$ pour tout $`D$ incluant $`G$ (lemme \ref{interp-weak}).
    \end{induction}
    
    \case{App}\quad
    \typenva

      \TAX{}
      {$\GD \seq f : F \quad \mualgo{F} = \Pi y : A. B : s$}

      {$E \sub A$}
      {$\GD \seq (f~e) : B [ e / y ]$}
      {}

    \[\ip{f[u/x]}{\Gr} `= \ip{f}{\GD}\ipux\]
    \[\ip{e[u/x]}{\Gr} `= \ip{e}{\GD}\ipux\]

    \begin{eqnarray*}
      \ip{(f~e)}{\GD}
      & = & (((\pi_F~\ip{f}{\GD})~(c_e~\ip{e}{\GD})) \\
      \text{ où } & & \\
      \pi_F & = & \coerce{\GD}{F}{(\Pi y : A.B)} \\
      c_e & = & \coerce{\GD}{E}{A}.
    \end{eqnarray*}
    
    On a la substitutivité du typage \ref{substitutive-typing} donc le
    jugement substitué est:

      {$\Gr \seq f[u/x] : F[u/x]$}
      {$\mualgo{F[u/x]} = \Pi y : A[u/x]. B[u/x] : s$}

      {$`G \seq E[u/x], A[u/x] : s$}
      {$E[u/x] \sub A[u/x]$}
      {$\Gr \seq (f~e)[u/x] : B' [ e[u/x] / y ]$}
      {}
    \end{prooftree}

      & = & \pi_{F[u/x]}[\ip{f'}{\Gr}]~c_{e'}[\ip{e'}{\Gr}] \\
      \text{ où } & & \\
      \pi_{F[u/x]} & = & \coerce{\Gr}{F[u/x]}{(\Pi y : A[u/x].B[u/x])} \\

    et $\subimpl{\Gr}{e}{E[u/x]}{A[u/x]}$ avec
    $d `= \pi_F\ipux$ et $e `= c_e\ipux$. Par unicité des
    coercions on en déduit que $d = \pi_{F[u/x]}$ et $e = c_{e'}$.

      \firsteq{\ip{f~e}{\GD}\ipux}

      {\eqdef}(\pi_F[\ip{f}{\GD}])~(c_e[\ip{e}{\GD}])\ipux 

      \step{Définition de la substitution}
      {=}{(\pi_F\ipux[\ip{f}{\GD}\ipux])~(c_e\ipux[\ip{e}{\GD}\ipux])}

      \step{Application de l'hypothèse d'induction pour les termes}
      {`=}{(\pi_F\ipux[\ip{f'}{\Gr}])~(c_e\ipux[\ip{e'}{\Gr}])}

      \step{Application de l'hypothèse d'induction pour les coercions}

On va maintenant étendre la relation de coercion aux contextes de
manière canonique.
\typenva
\begin{definition}[Coercion de contextes]
  \label{coercion-ctx}
  On définit inductivement la coercion de deux contextes de coercions
  algorithmiques par les règles suivantes:
  \begin{itemize}
  \item $[] \suba []$
  \item $(`G, x : T) \suba (`G', x : T')$ si $`G \suba `G'$ et $T \suba T'$.
  \end{itemize}
\end{definition}

De même pour les coercions explicites dérivées par le jugement $\subimpl{`G}{c}{T}{S}$.
\begin{definition}[Coercion explicites de contextes]
  \label{coercion-ctx-i}
  On définit inductivement la coercion de deux contextes de coercions
  explicites par les règles suivantes:
  \begin{itemize}
  \item $[] \suba []$
  \item $`r, c : (`G, x : T) \suba (`G', x : T')$ si $`r : `G \suba `G'$ et 
    $\subimpl{`G}{c}{T}{T'}$.
  \end{itemize}
\end{definition}

Clairement toute coercion de contexte algorithmique correspond à une
coercion de contexte explicite et vice-versa.

\begin{definition}[Extension de la substitution aux coercions de
  contextes]
  \label{subst-coercion-ctx}
  On définit la substitution d'une coercion de contexte inductivement:
  \begin{itemize}
  \item $t[[]] = t$
  \item $t[(`r, c) : (`G, x : T) \suba (`G', x : T')] = t[`r : `G \suba `G'][c[x]/x]$
  \end{itemize}
\end{definition}

\begin{lemma}[Stabilité par affaiblissement des coercions]
  \label{coerce-narrowing-coercion}
  Si $\subimpl{`G}{c}{T}{T'}$ et $\ip{T'}{`D} `=
  \ip{T'}{`G}[`r]$, alors pour tout $`D$ tel que $`r : `D \sub `G$,
  $\subimpl{`D}{c'}{T}{T'}$ et $c' `= c[`r]$ avec une dérivation de même
  taille.
\end{lemma}

\begin{proof}
  Par induction sur la dérivation de coercion:

  \begin{induction}
    \case{SubHnf} 
    On a $\subimpl{`G}{c}{\hnf{T}}{\hnf{T'}}$. Par induction, 
    $\subimpl{`D}{c[`r]}{\hnf{T}}{\hnf{T'}}$. 
    On peut donc dériver $\subimpl{`D}{c[`r]}{T}{T'}$ par \irule{SubHnf}.

    \case{SubConv}
    On a $T \eqbr T'$ et $c = \ctxdot$. 
    On peut donc dériver $\subimpl{`D}{c' = \ctxdot}{T}{T'}$,
    on a bien $c' `= c[`r]$.

    \case{SubProd}
    On a:
    \begin{prooftree}
      \BAX{}
      {$\subimpl{`G}{c_1}{C}{A}$}
      {$\subimpl{`G, x : C}{c_2}{B}{D}$}
      {$\subimpl{`G}{\lambda x : \ip{C}{`G}.~c_2[\ctxdot~c_1[x]]}
        {\Pi x : A.B}{\Pi x : C.D}$}
      {}
    \end{prooftree}

    Par induction on a $\subimpl{`D}{c_1[`r]}{C}{A}$. On peut définir
    la coercion $`s = `r, \ctxdot : (`D, x : C) \suba (`G, x : C)$ et obtenir par
    induction: $\subimpl{`D, x : C}{c_2'}{B}{D}$ avec $c_2' `= c_2[`s]$.
    
    On peut alors appliquer \irule{SubProd} pour obtenir:
    \[\subimpl{`D}
    {c' = \lambda x : \ip{C}{`D}.~c_2[`s][\ctxdot~c_1[`r][x]]}{\Pi x : A.B}{\Pi x : C.D}\]
    
    On a:
    \[\begin{array}{ll}
      \firsteq{(\lambda x : \ip{C}{`G}.~c_2[\ctxdot~c_1[x]])[`r]}
      
        \step{Définition de la substitution}
        {=}{\lambda x : \ip{C}{`G}[`r].(c_2[`r])[\ctxdot~(c_1[`r])[x]]}

        \step{Condition $\ip{T'}{`D} `= \ip{T'}{`G}[`r]$}
        {`=}{\lambda x : \ip{C}{`D}.(c_2[`r])[\ctxdot~(c_1[`r])[x]]}
        
        \step{Coercion identité dans $`s$}
        {=}{\lambda x : \ip{C}{`D}.(c_2[`s])[\ctxdot~(c_1[`r])[x]]}
      \end{array}
      \]
      
    \case{SubSigma}\quad
    \begin{prooftree}
      \BAX{}
      {$\subimpl{`G}{c_1}{A}{C}$}
      {$\subimpl{`G, x : A}{c_2}{B}{D}$}
      {$\subimpl{`G}{\pair{\ip{\Sigma x : C.D}{`G}}{c_1[\pi_1~\ctxdot]}{c_2[\pi_1~\ctxdot/x][\pi_2~\ctxdot]}}
        {\Sigma x : A.B}{\Sigma x : C.D}$}
      {}
    \end{prooftree}
    Par induction on a $\subimpl{`D}{c_1[`r]}{A}{C}$. On peut définir
    la coercion $`s = `r, \ctxdot : (`D, x : A) \suba (`G, x : A)$ et obtenir par
    induction:
    $\subimpl{`D, x : A}{c_2[`s]}{B}{D}$
    
    On peut alors appliquer \irule{SubSigma} pour obtenir:
    \[\subimpl{`D}
    {c' = \pair{\ip{\Sigma x : C.D}{`D}}{(c_1[`r])[\pi_1~\ctxdot]}{(c_2[`s])[\pi_1~\ctxdot/x][\pi_2~\ctxdot]}}
    {\Sigma x : A.B}{\Sigma x : C.D}\]
    
    On a:
    \[\begin{array}{ll}
      \firsteq{c[`r]}

      \step{Définition}
      {\eqdef}
      {(\pair{\ip{\Sigma x : C.D}{`G}}{c_1[\pi_1~\ctxdot]}{c_2[\pi_2~\ctxdot][\pi_1~\ctxdot/x]})[`r]}
      
      \step{Définition de la substitution, $x `; `r$}
      {=}{\pair{\ip{\Sigma x : C.D}{`G}[`r]}{c_1[`r][\pi_1~\ctxdot]}{c_2[`r][\pi_2~\ctxdot][\pi_1~\ctxdot/x]}}

      \step{Condition}
      {=}{\pair{\ip{\Sigma x : C.D}{`D}}{c_1[`r][\pi_1~\ctxdot]}{c_2[`r][\pi_2~\ctxdot][\pi_1~\ctxdot/x]}}

      \step{Coercion identité dans $`s$}
      {=}{\pair{\ip{\Sigma x :
            C.D}{`D}}{c_1[`r][\pi_1~\ctxdot]}{c_2[`s][\pi_2~\ctxdot][\pi_1~\ctxdot/x]}}

      \step{Définition}{\eqdef}{c'}
      \end{array} 
      \]
      
   \case{SubProof}\quad
   \begin{prooftree}
     \UAX{}
     {$\subimpl{`G}{d}{T}{U}$}
     {$\subimpl{`G}
       {c = \elt{\ip{U}{`G}}
         {\ip{(\lambda x : U.P)}{`G}}
         {d}
         {\ex{\ip{P}{`G, x : U}[d/x]}}}
       {T}{\mysubset{x}{U}{P}}$}
     {}
   \end{prooftree}

   Par induction, on a $\subimpl{`D}{d[`r]}{T}{U}$. On peut donc dériver:
   \[\subimpl{`D}{c' = \elt{\ip{U}{`D}}
     {\ip{(\lambda x : U.P)}{`D}}
     {d[`r]}
     {\ex{\ip{P}{`D, x : U}[d[`r]/x]}}}
   {T}{\mysubset{x}{U}{P}}\]
   
   On a bien $c[`r] `= c'$, car:
   \begin{equations}
     \firsteq{c[`r]}

     \step{Définition}
     {\eqdef}{(\elt{\ip{U}{`G}}
       {\ip{(\lambda x : U.P)}{`G}}
       {d}
       {\ex{\ip{P}{`G, x : U}[d/x]}})[`r]}

     \step{Condition $\ip{\mysubset{x}{U}{P}}{`D} `= \ip{\mysubset{x}{U}{P}}{`G}[`r]$}
     {`=}{\elt{\ip{U}{`D}}
       {\ip{(\lambda x : U.P)}{`D}}
       {d[`r]}
       {\ex{\ip{P}{`G, x : U}[d/x][`r]}}}
     
     \step{Indifférence aux preuves}
     {=_\sigma}{\elt{\ip{U}{`D}}
       {\ip{(\lambda x : U.P)}{`D}}
       {d[`r]}
       {\ex{\ip{P}{`D, x : U}[d[`r]/x]}}}
     
     \step{Définition}
     {\eqdef}{c'}
   \end{equations}

   \case{SubSub} Direct par induction.
  \end{induction}
\end{proof}

\begin{lemma}[Stabilité par affaiblissement]
  \label{coerce-narrowing-term}
  Si $\talgo{`G}{t}{T}$ alors pour tout $`D$,
  $`r : `D \sub `G$, $\talgo{`D}{t}{T'}$ et il existe $`a$, $\subimpl{`D}{`a}{T'}{T}$ avec
  $\ip{t}{`G}[`r] `= `a[\ip{t}{`D}]$.
\end{lemma}

\begin{proof}
  La première partie du lemme se déduit par applications répétées du
  lemme de restriction \ref{narrowing-a}.
  
  On va utiliser à plusieurs reprises le résultat suivant:
  Si une prémisse du jugement sur lequel on fait la récurrence est de la
  forme $\timpl{`G}{T}{s}$, alors on peut appliquer l'hypothèse
  d'induction pour obtenir $\timpl{`D}{T}{s'}$ avec $`a$ tel que
  $\subimpl{`D}{`a}{s}{s'}$ et $\ip{T}{`G}[`r] `=
  `a[\ip{T}{`D}]$. Or comme $s$ est une sorte, on a $s' = s$ et $`a =
  \ctxdot$. On a donc $\ip{T}{`G}[`r] `= \ip{T}{`D}$. 

  Par induction sur la dérivation de $t$.
  
  \begin{induction}
    \case{Var} 
    On a:
    \typenva
    \begin{prooftree}
      \BAX{Var}
      {$\wf `G$}
      {$y : T `: `G$}
      {$`G \seq y : T$}
      {}
    \end{prooftree}    
    
    Par définition de la coercion de contextes, il existe $c : (T' \sub
    T) `: `r$. Soit $`a = c$, on a bien: $\ip{y}{`G}[`r] = c[y] =
    `a[\ip{y}{`D}]$.

      {$E \sub A $}
      {$\talgo{`G}{(f~e)}{B[ e / y ]}$}
      {}
    \end{prooftree}
    
    et donc par induction, $\talgo{`D}{f}{F'}$ avec
    $\subimpl{`D}{`a_f}{F'}{F}$ et
    $\talgo{`D}{e}{E'}$ avec $\subimpl{`D}{`a_e}{E'}{E}$.

    Par définition de l'interprétation, 
    \begin{eqnarray*}
      \ip{(f~e)}{`G}
      & = & (((\pi_F[\ip{f}{`G}])~(c_e[\ip{e}{`G}])) \\
      \text{ où } & & \\
      \pi_F & = & \coerce{`G}{F}{\Pi y : A.B} \\
      c_e & = & \coerce{`G}{E}{A}.
    \end{eqnarray*}
    
    De même:
    \begin{eqnarray*}
      \ip{f~e}{`D}
      & = & (\pi_{F'}[\ip{f'}{`D}])~(c_{e'}[\ip{e'}{`D}]) \\
      \text{ où } & & \\
      \pi_{F'}  & = & \coerce{`D}{F'}{\Pi y : A'.B'} \\
      c_{e'} & = & \coerce{`D}{E'}{A'}
    \end{eqnarray*}

    On a $\subimpl{`G}{\pi_F}{F}{\Pi y : A.B}$, donc
    par le lemme \ref{coerce-narrowing-coercion} on obtient
    $\subimpl{`D}{\pi_F[`r]}{F}{\Pi y : A.B}$. On peut appliquer ce
    lemme puisque $\ip{\Pi y : A.B}{`D} `= \ip{\Pi y : A.B}{`G}[`r]$ par
    induction ($`a$ est nécessairement l'identité).

    On a donc les coercions suivantes dans l'environnement $`D$:

      F\ar[rr]_{\pi_F[`r]} & & 
      \Pi y : A.B \\
      F'\ar[u]^{`a_f}\ar[rr]^{\pi_{F'}} & &
      {\Pi y : A'.B'}\ar@{-->}[u]^{c_f}}

    Par symétrie et transitivité de la coercion, on en déduit qu'il
    existe $c_f$, $\subimpl{`D}{c_f}{\Pi y : A'.B'}{\Pi y : A.B}$.
    La coercion $c_f$ est nécessairement de la forme:
    $\lambda y : \ip{A}{`D}.c_2[\ctxdot~c_1[y]]$ où 
    $\subimpl{`D}{c_1}{A}{A'}$ et $\subimpl{`D, x : A}{c_2}{B'}{B}$.

    $\subimpl{`D}{c_e[`r]}{E}{A}$, de même que précédemment on obtient

      A\ar[d]^{c_1} \\
      E'\ar[u]^{`a_e}
      \ar@{-->}[rr]^{c_{e'}} & & A'}

      \step{Définition de $c_f$ ($= \lambda x.c_2[\ctxdot~c_1[x]]$)}
      {=}{c_f[(\pi_{F'}[\ip{f}{`D}])~[c_e[`r]~[`a_e[\ip{e}{`D}]]]]}
      
      \step{Commutation du diagramme pour la fonction}
      {`=}{(\pi_F[`r][`a_f[\ip{f}{`D}]]~[c_e[`r][`a_e~\ip{e}{`D}]])}
      
      \step{Définitions de $`a_f$ et $`a_e$}
      {`=}{(\pi_F[`r][\ip{f}{`G}[`r]])~[c_e[`r][\ip{e}{`G}[`r]]]}

      \step{Définition de l'interprétation}{\eqdef}{\ip{f~e}{`G}[`r]}
      \vspace{0.2em}
    \end{equations}

      \Prod
    \end{prooftree}
    Par induction, $\talgo{`D}{T}{s_1}$ et $\talgo{`D, x : T}{U}{s_2}$
    (en utilisant une coercion identité ajoutée à $`r$).
    On peut donc appliquer \irule{Prod} pour obtenir le résultat désiré

    De même pour \irule{Sigma}, \irule{Subset}.

    \case{Abs} On a

      \Abs
    \end{prooftree}
    
    Par induction, $\talgo{`D}{\Pi x : T.U}{s}$ et il existe $U'$, 
    $\talgo{`D, x : T}{M}{U'}$ avec une coercion $\subimpl{`D, x :
      T}{`a'}{U'}{U}$ telle que $\ip{M}{`G, x : T}[`r, \ctxdot] `= `a'[\ip{M}{`D,
      x : T}]$.
    
    Soit $`a$ la coercion: $\subimpl{`D}{\lambda x :
      \ip{T}{`D}.`a'[\ctxdot~x]}{\Pi x : T.U'}{\Pi x : T.U}$.
    On a bien:

      {\eqdef}{`a[\lambda x : \ip{T}{`D}.\ip{M}{`D, x : T}]}

      \step{Définition de la coercion $`a$}
      {\eqdef}{\lambda x : \ip{T}{`D}.`a'[\ip{M}{`D, x : T}[x/x]]}

      {`=}{\lambda x : \ip{T}{`D}.\ip{M}{`G, x : T}[`r]}

    \begin{prooftree}
      \QAX{SumDep}
      {$`D \seq t : T'' \quad T'' \suba T$}
      {$`D \seq u : U''$}
      {$`D \seq \Sigma x : T.U : s$}
      {$`D \seq U'' : s \quad U'' \suba U[t/x]$}
      {$`D \seq \pair{\Sigma x : T.U}{t}{u} : \Sigma x : T.U$}
      {}
    \end{prooftree}

    \[\subimpl{`D}{`b_u}{U''}{U'}, `b_u[\ip{u}{`D}] `= \ip{u}{`G}[`r]\]

    \begin{prooftree}
      \PiLeftA
    \end{prooftree}

    Par induction, il existe $`a_t$, $\subimpl{`D}{`a_t}{S'}{S}$ et

      {=}{\pi_1~(`b[\mu_{S'}[\ip{t}{`D}]])}

      \step{Transitivité}
      {`=}{\pi_1~\mu_S[`r][`a_t[\ip{t}{`D}]]}

      \step{Hypothèse d'induction}
      {`=}{\pi_1~\mu_S[`r][\ip{t}{`G}[`r]]}

      \step{Définition de l'interprétation}
      {\eqdef}{\ip{\pi_1~t}{`G}[`r]}
    \end{equations}

      \PiRightA
    \end{prooftree}
    
    On nomme $\mu_S$ la coercion de $S$ à $\Sigma x : T.U$. 
    Par le lemme \ref{coerce-narrowing-coercion} on obtient:
    $\subimpl{`D}{\mu_S[`r]}{S}{\Sigma x : T.U}$. 

    On peut peut appliquer l'hypothèse d'induction sur
    $\talgo{`G}{t}{S}$ pour dériver:
    \begin{prooftree}
      \AXC{$\talgo{`D}{t}{S'}$}
      \AXC{$\mualgo{S'} = \Sigma x : T'.U'$}
      \BIC{$\talgo{`D}{\pi_1~t}{T'}$}
    \end{prooftree}

    Par induction, il existe $`a_t$, $\subimpl{`D}{`a_t}{S'}{S}$ et
    $`a_t[\ip{t}{`D}] = \ip{t}{`G}[`r]$.
    Par transitivité de la coercion, on sait qu'il existe une coercion
    de $S'$ à $\Sigma x : T.U$.
    On en déduit qu'il existe $T',U'$ tels que $\mualgo{S'} = \Sigma x :
    T'.U'$ et qu'il existe une coercion $`b$ telle que
    \[\subimpl{`D}{`b = \pair{\Sigma x :
        T.U}{c_1[\pi_1~\ctxdot]}{c_2[\pi_2~\ctxdot][\pi_1~\ctxdot/x]}}
    {\Sigma x : T'.U'}{\Sigma x : T.U}\]
    avec $`b `o \mu_{S'} `= \mu_S[`r] `o `a_t$.
    
    On a donc les coercions suivantes dans l'environnement $`D$:
    \[
    \xymatrix{
      S\ar[rr]_{\mu_S[`r]} & & 
      \Sigma x : T.U \\
      S'\ar[u]^{`a_t}
      \ar[rr]^{\mu_{S'}} & & \Sigma x : T'.U'\ar@{-->}[u]^{`b}}
    \]
    
    Par substitution pour les coercions, avec les dérivation de $\pi_1~t$
    et de $c_2$ on obtient:
    $\subimpl{`D}{c_2' `= c_2[\ip{\pi_1~t}{`D}/x]}{U'[\pi_1~t/x]}{U[\pi_1~t/x]}$.
    
    Soit $`a = c_2'$.    
    On a:
    \begin{equations}
      \firsteq{`a[\ip{\pi_2~t}{`D}]}
      
      \step{Définition de $c_2'$}
      {`=}{c_2[\ip{\pi_1~t}{`D}/x][\ip{\pi_2~t}{`D}]}
      
      \step{Définition de l'interprétation}
      {\eqdef}{c_2[\pi_1~\mu_{S'}[\ip{t}{`D}]/x][\ip{\pi_2~t}{`D}]}

      \step{Définition de l'interprétation}
      {\eqdef}{c_2[\pi_1~\mu_{S'}[\ip{t}{`D}]/x][\pi_2~\mu_{S'}[\ip{t}{`D}]]}

      \step{$\ctxdot `; \ip{t}{`D} `^ x `; \freevars{t}$}
      {\eqdef}{c_2[\pi_2~\mu_{S'}[\ip{t}{`D}]][\pi_1~\mu_{S'}[\ip{t}{`D}]/x]}

      \step{$\pi_2~`b = c_2[\pi_2~\ctxdot][\pi_1~\ctxdot/x]$}
      {=}{(\pi_2~`b)[\mu_{S'}[\ip{t}{`D}]]}

      \step{Définition de la substitution dans les contextes}
      {=}{\pi_2~(`b[\mu_{S'}[\ip{t}{`D}]])}

      \step{Transitivité}
      {`=}{\pi_2~\mu_S[`r][`a_t[\ip{t}{`D}]]}

      \step{Hypothèse d'induction}
      {`=}{\pi_2~\mu_S[`r][\ip{t}{`G}[`r]]}

      \step{Définition de l'interprétation}
      {\eqdef}{\ip{\pi_2~t}{`G}[`r]}
    \end{equations}

  \end{induction}
\end{proof}

On a besoin d'une spécialisation de ce lemme lorsque les types sont équivalents:
\begin{lemma}[Equivalence et interprétation]
  \label{coerce-narrowing-term-eqbr}
  Si $\talgo{`G}{U, U'}{s}$ et $U \eqbr U'$ alors:
  \begin{itemize}
  \item si $\talgo{`G, x : U, `D}{t}{T}$, alors $\talgo{`G, x : U', `D}{t}{T'}$ avec $T \eqbr T'$ et $\ip{t}{`G, x : U, `D} `= \ip{t}{`G, x : U', `D}$;
  \item si $\subalgo{`G, x : U, `D}{T}{T'}$ alors $\subalgo{`G, x : U', `D}{T}{T'}$.
  \end{itemize}
\end{lemma}

\begin{proof}
  Pour la première partie, la preuve suit le même schéma que la précédente. Il suffit de vérifier
  que l'on a toujours $`a `= \ctxdot$. Par exemple, dans le cas de l'application, on
  a $`a_f `= `a_e `= \ctxdot$. On en déduit que $c_f `= \ctxdot$ et
  $c_1 `= c_2 `= \ctxdot$. Clairement $c_2'' `= \ctxdot$. 
  
  Pour la seconde partie, c'est direct par induction en utilisant la
  première partie.
\end{proof}
