
\section*{Grammaire du langage}

\begin{figure}[h]
  \begin{displaymath}
    \begin{array}{lcl}
      `a, `b & \Coloneqq & x \\
      & | & \funml{}~x~:~`t "=>" `a \\
      & | & `a~`b \\ 
      & | & (`a,~`b) \\
      & | & (x \coloneqq `a,~`b~: `a) \\
      & | & 0 \ldots n \\
      & | & \letml~x = `a ~\inml~`b \\
      & | & \letml~(x_1, \ldots, x_n) = `a ~\inml~ `b     
    \end{array}
  \end{displaymath}
  \caption{Termes}
  \label{fig:gram-terms}
  
\end{figure}

\begin{figure}[h]

    \end{array}
