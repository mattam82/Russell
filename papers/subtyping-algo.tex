\newcommand{\subalgo}[4]{#1 \typea #3 \suba #4}
\newcommand{\SubConvA}[1][\Gamma]{
\UAX{SubConv}
{$T \eqbr U$}
{$\subalgo{#1}{\ctxdot}{T}{U}$}
{$T = \hnf{T} `^ T "/=" \Pi, \Sigma, \{|\} `^ U = \hnf{U}$}
}

\newcommand{\SubConvAs}[1][\Gamma]{
\UAX{SubConv}
{$T \eqbr U$}
{$\subalgo{#1}{\ctxdot}{T}{U}$}
{} 
}

\newcommand{\SubHnfA}[1][\Gamma]{
\UAX{SubHnf}
{$\subalgo{#1}{c}{\hnf{T}}{\hnf{U}}$}
{$\subalgo{#1}{c}{T}{U}$}
{$T "/=" \hnf{T} `V U "/=" \hnf{U}$} 
}

\newcommand{\SubProdA}[1][\Gamma]{%
  \BAX{SubProd}
  {$\subalgo{#1}{c_1}{U}{T} : s$}
  {$\subalgo{#1, x : U}{c_2}{V}{W}$}  
  {$\subalgo{#1}{\lambda x : \ip{U}{#1}. c_2[\ctxdot~(c_1[x])]}
    {\Pi x : T.V}{\Pi x : U.W}$}
  {}
}

\newcommand{\SubSigmaA}[1][\Gamma]{%
\BAX{SubSigma}
{$\subalgo{#1}{c_1}{T}{U}$}
{$\subalgo{#1, x : T}{c_2}{V}{W}$}
{$\subalgo{#1}{\pair{\ip{\Sigma x : U.W}{`G}}{c_1[\pi_1~\ctxdot]}{c_2[\pi_2~\ctxdot][\pi_1~\ctxdot/x]}}
  {\Sigma x : T. V}{\Sigma x : U. W}$}
{}
}

\newcommand{\SubSubA}[1][\Gamma]{%
\UAX{SubSub}
{$\subalgo{#1}{c}{U}{T}$}
{$\subalgo{#1}{c[\eltpit~\ctxdot]}{\mysubset{x}{U}{P}}{T}$}
{$T = \hnf{T} `^ \hnf{T} "/=" \{|\}$}
}

\newcommand{\SubSubAs}[1][\Gamma]{%
\UAX{SubSub}
{$\subalgo{#1}{c}{U}{T}$}
{$\subalgo{#1}{c[\eltpit~\ctxdot]}{\mysubset{x}{U}{P}}{T}$}
{}
}
   
\newcommand{\SubProofA}[1][\Gamma]{%
  \BAX{SubProof}
  {$\subalgo{#1}{c}{T}{U}$}
  {$\talgo{#1}{\mysubset{x}{U}{P}}{\Set}$}
  {$\subalgo{#1}
    {\elt{\ip{U}{#1}}{\ip{\lambda x : U.P}{#1}}{c}
      {\ex{\ipG{#1}}{\ip{P}{#1, x : U}[c/x]}}}
  {T}{\mysubset{x}{U}{P}}$}
{$T = \hnf{T}$}
}
