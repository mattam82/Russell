\def\SubConv{
\BAX{Sub-Conv}
{$`G \seq T, U : s$}
{$T \eqbi U$}
{$`G \seq T \sub U$}
{} 
}

\def\SubProd{
  \BAX{Sub-Prod}
  {$`G \seq U \sub T$} %"<|-|>"
  {$`G, x : U \seq V \sub W$}
  {$`G \seq \Pi x : T.V \sub \Pi x : U.W$}
  {}
}

\def\SubSigma{
  \BAX{Sub-Sigma}
  {$`G \seq T \sub U$}
  {$`G \seq V \sub W$}
  {$`G \seq \Sigma x : T. V \sub \Sigma y : U. W$}
  {}
}

\def\SubLeft{
  \BAX{Sub-Left}
  {$`G \seq U \sub V$}
  {$`G \type P : \Pi x : U. \Prop$}
  {$`G \seq \subset{x}{U}{P} \sub V$}
  {}
}
    
\def\SubRight{
  \UAX{Sub-Right}
  {$`G \seq T \sub U$}
                                %{$`G \judgetype h : P~p$}
  {$`G \seq T \sub \subset{x}{U}{P}$}
  {}
}
