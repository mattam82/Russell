\def\SubConv{
  \UAX{SubConv}
%  {$`G \type T, U : s$}
  {$`G \type T \eqbr U : s$}
  {$`G \type T \sub U : s$}
  {} 
}

\def\SubRefl{
  \UAX{SubRefl}
  {}
  {$S \sub S$}
  {}
}

\def\SubTrans{
\BAX{SubTrans}
{$`G \type S \sub T : s$}
{$`G \type T \sub U : s$}
{$`G \type S \sub U : s$}
{}
} 

\def\SubSym{
\UAX{SubSym}
{$`G \type U \sub T : s$}
{$`G \type T \sub U : s$}
{}
} 

\def\SubProd{
\BAX{SubProd}
{$`G \type U \sub T : s_1$} %"<|-|>"
{$`G, x : U \type V \sub W : s_2$}
{$`G \type \Pi x : T.V \sub \Pi x : U.W : s_2$}
{}
%{$(s_1, s_2) `: \PTSrules$}
}

\def\SubSigma{
  \BAX{SubSigma}
  {$`G \type T \sub U : s$}
  {$`G, x : T \type V \sub W : s$}
  {$`G \type \Sigma x : T. V \sub \Sigma y : U. W : s$}
  {$s `:  \setprop$}
}

\def\SubProof{
  \BAX{SubProof}
  {$`G \type U \sub V : \Set$}
  {$`G, x : V \type P : \Prop$}
  {$`G \type U \sub \mysubset{x}{V}{P} : \Set$}
  {}
}

\def\SubSub{
  \BAX{SubSub}
  {$`G \type U \sub V : \Set$}
  {$`G, x : U \type P : \Prop$}
  {$`G \type \mysubset{x}{U}{P} \sub V : \Set$}
  {}
}
