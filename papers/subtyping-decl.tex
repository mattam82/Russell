\def\SubConv{
\UAX{SubConv}
{$T \eqbi U$}
{$T \sub U$}
{} 
}

\def\SubRefl{
  \UAX{SubRefl}
  {}
  {$S \seq S$}
  {}
}

\def\SubTrans{
\BAX{SubTrans}
{$S \seq T$}
{$T \seq U$}
{$S \seq U$}
{}
} 

\def\SubProd{
\BAX{SubProd}
{$U \seq T$} %"<|-|>"
{$V \seq W$}
{$\Pi x : T.V \seq \Pi x : U.W$}
{}
}

\def\SubSigma{
  \BAX{SubSigma}
  {$T \seq U$}
  {$V \seq W$}
  {$\Sigma x : T. V \seq \Sigma y : U. W$}
  {}
}

\def\SubProof{
  \UAX{SubProof}
  {$U \seq V$}
  {$U \seq \subset{x}{V}{P}$}
  {}
}

\def\SubSub{
  \UAX{SubSub}
  {$U \seq V$}
  {$\subset{x}{U}{P} \seq V$}
  {}
}
