\def\SubConv{
\UAX{Sub-Conv}
{$T \eqbi U$}
{$T \sub U$}
{} 
}

\def\SubRefl{
  \UAX{Sub-Refl}
  {}
  {$S \seq S$}
  {}
}

\def\SubTrans{
\BAX{Sub-Trans}
{$S \seq T$}
{$T \seq U$}
{$S \seq U$}
{}
} 

\def\SubProd{
\BAX{Sub-Prod}
{$U \seq T$} %"<|-|>"
{$V \seq W$}
{$\Pi x : T.V \seq \Pi x : U.W$}
{}
}

\def\SubSigma{
  \BAX{Sub-Sigma}
  {$T \seq U$}
  {$V \seq W$}
  {$\Sigma x : T. V \seq \Sigma y : U. W$}
  {}
}

\def\SubProof{
  \UAX{Sub-Proof}
  {$U \seq V$}
  {$U \seq \subset{x}{V}{P}$}
  {}
}

\def\SubSub{
  \UAX{Sub-Subset}
  {$U \seq V$}
  {$\subset{x}{U}{P} \seq V$}
  {}
}
