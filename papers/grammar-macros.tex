\newcommand{\termgrammar}
{$\begin{array}{lcl}
    `a & \Coloneqq & x \\
    & | & \funml{}~x~:~`t "=>" `a \\
    & | & `a~`a \\ 
    & | & (`a,~`a) \\
    & | & (x \coloneqq `a,~`a~: `t) \\
    & | & 0 \ldots n \\
    & | & \letml~x = `a ~\inml~`a \\
    & | & \letml~(x_1, \ldots, x_n) = `a ~\inml~ `a \\
    & | & \ifml~`a~\thenml~`a~\elseml~`a
  \end{array}$}
