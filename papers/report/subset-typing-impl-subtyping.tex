\begin{lemma}[Unicité de la coercion]
  \label{coerce-unicity}
  Si $`G \typea T, U : s$ alors si $\subimpl{`G}{c}{T}{U}$ et
  $\subimpl{`G}{c'}{T}{U}$ alors $c = c'$.
\end{lemma}
\begin{proof}
  Par simple inspection des règles, on remarque qu'une seule règle
  s'applique suivant la forme de $T$, sauf si $T$ et $U$ sont des types
  sous-ensemble.
  On montre donc la confluence des deux règles \irule{SubSub} et \irule{SubProof}:
  \begin{prooftree}
    \AXC{$\subimpl{`G}{c}{T'}{U'}$}
    \UIC{$\subimpl{`G}{c[\eltpit~\ctxdot]}{\mysubset{x}{T'}{P}}{U'}$}
    \UIC{$\subimpl{`G}{c'}{T = \mysubset{x}{T'}{P}}{U = \mysubset{x}{U'}{Q}}$}
  \end{prooftree}
  avec \[c' = \elt{\ip{U'}{`G}}{\ip{\lambda x :
      U'.Q}{`G}}{c[\eltpit~\ctxdot]}
  {\ex{\ip{Q}{`G, x : U'}[c[\eltpit~\ctxdot]/x]}}\]
  D'autre part:
  
  \begin{prooftree}
    \AXC{$\subimpl{`G}{c}{T'}{U'}$}
    \UIC{$\subimpl{`G}{c''}{T'}{\mysubset{x}{U'}{Q}}$}
    \UIC{$\subimpl{`G}{c''[\eltpit~\ctxdot]}{T = \mysubset{x}{T'}{P}}{U}$}
  \end{prooftree}
  avec \[c'' = \elt{\ip{U'}{`G}}{\ip{\lambda x :
      U'.Q}{`G}}{c}{\ex{\ip{Q}{`G, x : U'}[c/x]}}\]

  Clairement, $c''[\eltpit~\ctxdot] = 
  \elt{\ip{U'}{`G}}{\ip{\lambda x :
      U'.Q}{`G}}{c[\eltpit~\ctxdot]}{\ex{\ip{Q}{`G, x :
        U'}[c[\eltpit~\ctxdot]/x]}}
  = c'$.

  La confluence locale suffit puisqu'il n'est pas possible d'appliquer
  d'autres règles que ces deux là si ces deux là sont applicables.
\end{proof}

On peut donc raisonner comme ceci: si l'on parvient à construire une
dérivation de $\subimpl{`G}{c}{T}{U}$ toute autre dérivation de $T \suba U$
donne le même terme de coercion.

\begin{lemma}[Réflexivité de la coercion]
  \label{coerce-refl}
  Si $\timpl{`G}{A}{s}$ alors il existe $c$, $\subimpl{`G}{c}{A}{A}$ et
  $c `= \ctxdot$.
\end{lemma}
\begin{proof}
  Par induction sur le nombre de constructeurs $\Pi, \Sigma, \{|\}$ dans
  la forme normale de $A$.

  S'il $\nf{A}$ n'a pas de constructeur $\Pi, \Sigma, \{|\}$ en tête,
  alors $\hnf{A}$ n'en a pas en tête et l'on peut directement appliquer
  \irule{SubConv}.
  
  Sinon, on va appliquer la ou les règles correspondant au constructeur
  en tête. Le cas le plus intéressant est si $\hnf{A}$ est de la forme
  $\mysubset{x}{U}{P}$.
  Alors on a la dérivation:

  \begin{prooftree}
    \AXC{$\subimpl{`G}{c' `= \ctxdot}{U}{U}$}
    \UIC{$\subimpl{`G}{d = c'[\eltpit~\ctxdot]}{\mysubset{x}{U}{P}}{U}$}
    \UIC{$\subimpl{`G}{c = \elt{\ip{U}{`G}}{\ip{\lambda x : U.P}{`G}}{d}
        {\ex{\ip{P}{`G, x : U}[d/x]}}}{\hnf{A} = \mysubset{x}{U}{P}}{\mysubset{x}{U}{P}}$}
    \UIC{$\subimpl{`G}{c}{A}{A}$}
  \end{prooftree}

  On obtient la dérivation de $c'$ par induction, puisque $\nf{U}$
  contient strictement moins de constructeurs que $\nf{A}$.

  On a: 
  \begin{eqnarray*}
    c & \eqdef & \elt{\ip{U}{`G}}{\ip{\lambda x : U.P}{`G}}{d}
    {\ex{\ip{P}{`G, x : U}[d/x]}} \\
    & \eqdef & \elt{\ip{U}{`G}}{\ip{\lambda x : U.P}{`G}}{(\eltpit~\ctxdot)}
    {\ex{\ip{P}{`G, x : U}[\eltpit~\ctxdot/x]}} \\
    & =_{\sigma} & \elt{\ip{U}{`G}}{\ip{\lambda x : U.P}{`G}}{(\eltpit~\ctxdot)}
    {(\eltpip~\ctxdot)} \\
    & =_{\rho} & \ctxdot
  \end{eqnarray*}

  On utilise ici l'indifférence aux preuves ($\sigma$). Comme $(\eltpip~\ctxdot)$ est
  de type $\ip{P}{`G, x : U}[\eltpit~\ctxdot/x]$ dans un contexte où $\ctxdot$
  est de type $\ip{\mysubset{x}{U}{P}}{`G}$, on peut remplacer directement
  l'existentielle par ce terme. En pratique, ces preuves pourront être
  directement déchargées par l'assistant de preuve.
\end{proof}

\begin{lemma}[Coercion et sortage]
  \label{coerce-sorting}
  Si $\talgo{`G}{T}{s_1}$, $\talgo{`G}{U}{s_2}$ et
  $\subalgo{`G}{T}{U}$ alors $s_1 = s_2$ et la taille des dérivations de
  sortage $\talgo{`G}{T,U}{s}$ est plus petite que la taille de la dérivation de coercion.
\end{lemma}
\begin{proof}
  Par induction sur la dérivation de coercion.
  \begin{induction}
    \casetwo{SubConv}{SubHnf} Trivial.
    
    \case{SubProd} Par induction et application de la règle
    \irule{Prod} ; on repose sur le fait que la relation $\mathcal{R}$
    est fonctionnelle.

    \case{SubSigma} Direct par induction et application de la règle
    \irule{Sigma}.

    \case{SubSub}
    \begin{prooftree}
      \AXC{$\subalgo{`G}{T'}{U}$}
      \UIC{$\subalgo{`G}{\mysubset{x}{T'}{P}}{U}$}
    \end{prooftree}
    Si $\talgo{`G}{\mysubset{x}{T'}{P}}{s_1}$, alors $s_1 = \Set$ et $\talgo{`G}{T'}{\Set}$.
    Donc par induction, $\talgo{`G}{U}{\Set = s_2}$.

    \case{SubProof}
    \begin{prooftree}
      \AXC{$\subalgo{`G}{T}{U'}$}
      \UIC{$\subalgo{`G}{T}{\mysubset{x}{U'}{P}}$}
    \end{prooftree}
    
    Par inversion du jugement de typage
    $\talgo{`G}{\mysubset{x}{U'}{P}}{s_2}$, on a $s_2 = \Set$ et $\talgo{`G}{U'}{\Set}$,
    donc par application de l'hypothèse d'induction,
    $\talgo{`G}{T}{\Set}$.
  \end{induction}
\end{proof}

Ce lemme montre que nous avons suffisament de conditions de sortes dans
nos règles pour assurer qu'en n'importe quel point d'une dérivation de
coercion commençant par deux types bien sortés, nous travaillons sur des
types de même sorte, et que nous pouvons appliquer nos hypothèses
d'induction mutuelle aux jugement de sortage correspondants.

\begin{lemma}[Coercion et formes normales de tête]
  \label{coerce-hnf}
  Si $`G \typec c : T \suba U$ alors $`G \typec c' : \hnf{T}~\suba
  \hnf{U}$ avec $c = c'$ est dérivable par une dérivation plus
  petite ou égale.
\end{lemma}
\begin{proof}
  Par idempotence de la mise en forme normale de tête, on a la même
  dérivation dans le cas où la dernière règle appliquée était
  \irule{SubHnf}, sinon c'est trivial.
\end{proof}

On va maintenant généraliser la réflexivité aux termes convertibles.

\begin{lemma}[Coercion de termes convertibles]
  \label{coerce-eqbr-eq-id}
  Si $\talgo{`G}{T, U}{s}$, $T \eqbr U$ alors il existe $c$,
  $\subimpl{`G}{c}{T}{U}$ avec $c `= \ctxdot$.
\end{lemma}
\begin{proof}
  Par induction sur le nombre de constructeurs $\Pi, \Sigma, \{|\}$ dans
  les formes normales de $T$ et $U$.
  
  \begin{induction}  
  \casenil Si $\nf{T}$ n'a pas pour symbole de tête, $\Pi, \Sigma$ ou
    $\{|\}$, alors \irule{SubConv} est la seule règle applicable et on
    a $c = \ctxdot$. 

  \casenil Si $\hnf{T} = \Pi y : A.B$, alors $\hnf{U} = \Pi y : A'.B'$ avec $A
    \eqbr A'$, $B \eqbr B'$. Par induction, $\subimpl{`G}{c_1 `= \ctxdot}{A'}{A}$
    et $\subimpl{`G, x : A'}{c_2 `= \ctxdot}{B}{B'}$ ($\nf{A}, \nf{A'}, \nf{B}, \nf{B'}$ sont des
    sous-termes stricts de $\nf{T}$ et $\nf{U}$). On peut donc dériver: 
    $\subimpl{`G}{\lambda x : \ip{A'}{`G}. c_2[\ctxdot~c_1[x]]}
    {\Pi y : A.B}{\Pi y : A'.B'}$. Puis par application de
    \irule{SubHnf}, la coercion de $T$ à $U$.
        
    On a donc:
    \begin{eqnarray*}
      c & \eqdef & \lambda x : \ip{A'}{`G}. c_2[\ctxdot~(c_1[x]) \\
      & `= & \lambda x : \ip{A'}{`G}. \ctxdot~x \\
      & =_{\eta} & \ctxdot
    \end{eqnarray*}
    
  \casenil Si $\hnf{T} = \Sigma y : A.B$ alors $\hnf{U} = \Sigma y : A'.B'$ avec $A
    \eqbr A'$, $B \eqbr B'$. Par induction, $\subimpl{`G}{c_1 `= \ctxdot}{A}{A'}$
    et $\subimpl{`G, y : A}{c_2 `= \ctxdot}{B}{B'}$ ($\nf{A},
    \nf{A'}, \nf{B}, \nf{B'}$ sont des
    sous-termes stricts de $\nf{T}$ et $\nf{U}$). On peut donc dériver: 
    $\subimpl{`G}{\pair{\ip{\Sigma y :
          A'.B'}{`G}}{c_1[\pi_1~\ctxdot]}{c_2[\pi_2~\ctxdot][\pi_1~\ctxdot/y]}}    
    {\Sigma y : A.B}{\Sigma y : A'.B'}$. Puis par application de
    \irule{SubHnf}, la coercion de $T$ à $U$.

    On a donc:
    \begin{eqnarray*}
      c & \eqdef & \pair{\ip{\Sigma y : A'.B'}{`G}}
      {c_1[\pi_1~\ctxdot]}
      {c_2[\pi_2~\ctxdot][\pi_1~\ctxdot/y]} \\
      & `= & 
      \pair{\ip{\Sigma y : A'.B'}{`G}}
      {\pi_1~\ctxdot}
      {(\pi_2~\ctxdot)[\pi_1~\ctxdot/y]} \\
      & = & 
      \pair{\ip{\Sigma y : A'.B'}{`G}}
      {\pi_1~\ctxdot}
      {\pi_2~\ctxdot} \\
      & =_\rho & \ctxdot
    \end{eqnarray*}
    
    La substitution est inutile puisque dans un contexte où $\ctxdot :
    \Sigma y : A.B$, $y `; \pi_2~\ctxdot$.

  \casenil Le cas des sous-ensembles est un peu différent.
    Si $\hnf{T} = \mysubset{x}{T'}{P}$ alors $\hnf{U} =
    \mysubset{x}{U'}{P'}$ avec $T' \eqbr U'$ et $P \eqbr P'$.
    La dérivation va avoir la forme suivante:
    \begin{prooftree}
      \AXC{$\subimpl{`G}{c' `= \ctxdot}{T'}{U'}$}
      \UIC{$\subimpl{`G}{d = c'[\eltpit~\ctxdot]}{\mysubset{x}{T'}{P}}{U'}$}
      \UIC{$\subimpl{`G}{c = \elt{\ip{U'}{`G}}{\ip{\lambda x : U'.P'}{`G}}{d}
          {\ex{\ip{P'}{`G, x : U'}[d/x]}}}{\mysubset{x}{T'}{P}}{\mysubset{x}{U'}{P'}}$}
      \UIC{$\subimpl{`G}{c}{T}{U}$}
    \end{prooftree}
    
    On a donc $d = c'[\eltpit~\ctxdot] `= \eltpit~\ctxdot$.

    On peut vérifier:
    \begin{eqnarray*}
      c & \eqdef & \elt{\ip{U'}{`G}}{\ip{\lambda x : U'.P'}{`G}}{d}
      {\ex{\ip{P'}{`G, x : U'}[d/x]}} \\
      & `= & \elt{\ip{U'}{`G}}{\ip{\lambda x : U'.P'}{`G}}{(\eltpit~\ctxdot)}
      {\ex{\ip{P'}{`G, x : U'}[\eltpit~\ctxdot/x]}} \\
      & =_\sigma & \elt{\ip{U'}{`G}}{\ip{\lambda x : U'.P'}{`G}}{(\eltpit~\ctxdot)}
      {(\eltpip~\ctxdot)} \\
      & =_\rho & \ctxdot
    \end{eqnarray*}    
    
    On fait ici un usage très libéral de l'indifférence aux preuves. En
    effet nous ne pouvons pas encore montrer que $\eltpip~\ctxdot :
    \ip{P'}{`G, x : U'}[\eltpit~\ctxdot/x]$ puisqu'on sait seulement
    que dans le contexte où $\ctxdot : \ip{\mysubset{x}{T'}{P}}{`G}$,
    $\eltpip~\ctxdot : \ip{P}{`G, x : T'}[\eltpit~\ctxdot/x]$. Montrer
    que les interprétations de $P$ et $P'$ sont équivalentes requiert la
    stabilité de la convertibilité par interprétation, ce que nous
    montrerons plus tard.
  
  \end{induction}
\end{proof}
