\begin{frame}
  \frametitle{Types sous-ensemble}
  
  \begin{block}{Définition}
    $\subset{x}{T}{P}$ est le type des termes de type $T$ vérifiant la
    propriété $P$. 
  \end{block}
  
  \begin{itemize}
  \item Très utilisé pour spécifier ($\sref{div}, \sref{rev}$) ;
  \item Séparation objet / propriété ;
  \end{itemize}

  \begin{block}{Idée} 
    Un objet de type $T$ peut être vu comme un objet de
    type $\subset{x}{T}{P}$ pour toute propriété $P$ bien
    formée et vice-versa.
  \end{block}

\end{frame}
