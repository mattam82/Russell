\begin{frame}
  \frametitle{Types sous-ensemble}
  
  \begin{block}{Définition}
    $\mysubset{x}{T}{P}$ est le type des termes de type $T$ vérifiant la
    propriété $P$. 
  \end{block}
  
  \begin{itemize}
  \item Très utilisé pour spécifier ($\sref{div}, \sref{map}$), présent
    en mathématiques ;    
  \item Séparation objet / propriété ;
  \end{itemize}

  \begin{block}{Idée} 
    Un objet de type $T$ peut être vu comme un objet de
    type $\mysubset{x}{T}{P}$ % pour toute propriété $P$ correcte
    et vice-versa.
  \end{block}
  
\end{frame}

\begin{frame}[t,label=sysalgo]
  \frametitle{Système algorithmique}
  \begin{block}{Objectif}
    Extraire un algorithme du système formel. Nouveau jugement de
    typage $\typea$ qu'on veut \emph{décidable}.
  \end{block}
  
  \begin{itemize}
  \item<2-> Intégration de la coercion à l'application 
    \hyperlink{typingdecl<3>}{\beamergotobutton{}} ;

    \hyperlink{subtypingdecl<3>}{\beamergotobutton{}} ;
  \end{itemize}

  \only<2>{\TAX{App}
    {$`G \seq f : T \quad \mu(T) `= \Pi x : V. W$}
    {$`G \seq u : U$}
    {$U \subd V $}
    {$`G \seq (f u) : W [ u / x ]$}
    {}
    \DP
    
    \begin{eqnarray*}
      \mu~\mysubset{x}{U}{P} & "=>" & \mu~U \\
      \mu~x                & "=>" & x
    \end{eqnarray*}
  }
  
  \only<4>{

      $`G \typed t : T$ \ssi{} il existe $U$ tel que $`G \typea t : U$
      avec $U \suba T$ et $U \suba T$ \ssi{} $U \subd T$.

  \frametitle{Interprétation}
  
  \begin{center}
    $\ip{t}{`G} = t', T'$. On note $\ipt{t}{`G} = t'$ et $\ipT{t}{`G} = T'$.
  \end{center}
  
  \only<2->{
  Exemple: 
  \[\begin{array}{lcll}
    \ip{f~u}{`G} 
    & = & \letml~(f', `t) = \ip{f}{`G}~\inml & \\
    & & \letml~(\pi, \Pi x : V.W) = \muimpl(`t)~\inml & \\
    & & \letml~(u', U') = \ip{u}{`G}~\inml & \\
    & & \letml~c = \sref{coerce}~U'~V~\inml & \\
    & & (\pi~f')~(c~u'), W[c~u'/x] & 
  \end{array}\]
 
    \setboolean{displayLabels}{false}
    \AppA\DP
  }
   
  % \item Correspondance code écrit/extrait.
\end{frame}

\begin{frame}
  \frametitle{Propriétés}
  
  \begin{conjecture}[Correction de la réécriture]
    Si $`G \typea t : U$ alors $\ipG{`G} \typec \ipt{t}{\iG} : \ipt{T}{\iG}$.
  \end{conjecture}
  
  \begin{lemma}[Substitutivité]
    $\ipt{W}{\ipG{`G, x : V}}[c~\ipt{u}{\iG}/x] \eqbi \ipt{W[u/x]}{\iG}$
  \end{lemma}
  
  \begin{center}
    $"=>"$ Commutation des coercions
  \end{center}
\end{frame}
